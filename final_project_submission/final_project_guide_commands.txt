# Commands to Build and Run the AGE Game Engine Project

# Step 3: Build all three games
make

# Step 4: Run Game 1 - Flappy Bird
cd game-1-files
./flappy_bird
# Controls: SPACE to start, SPACE or W to flap, Q to quit, R to restart

# Step 5: Run Game 2 - Space Invaders
cd ../game-2-files
./space_invaders
# Controls: SPACE to start, A/D to move, SPACE/W/MB1 to fire, Q to quit, R to restart

# Step 6: Run Game 3 - extra
cd ../game-3-files
./game_3_extra
# Controls: SPACE to start, A/D to move, Q to quit, R to restart
